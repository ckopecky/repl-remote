Backend flow
async function processContact(input: ContactInput) {
  const normalized = normalizeContact(input);

  const decision = await triageWithMelanie(normalized);

  if (decision.decision === "absolute_no") {
    await saveDecision(normalized, decision);
    return;
  }

  const signal = await createGTMSignal(normalized, decision);

  if (decision.decision === "reach_out_now") {
    await routeToCampaign(signal, decision);
  }
}
